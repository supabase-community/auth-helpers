  • default(context: GetServerSidePropsContext<ParsedUrlQuery, PreviewData> | { req: NextApiRequest }, cookieOptions?: CookieOptions): SupabaseClient
    • Defined in src/nextjs/utils/supabaseServerClient.ts:30

    This is a helper method to wrap your SupabaseClient to inject a user's access_token to make use of RLS on the server side.

    import { supabaseServerClient } from '@supabase/supabase-auth-helpers/nextjs';

    export async function getServerSideProps(context) {
      // Run queries with RLS on the server
      const { data } = await supabaseServerClient(context)
        .from('test')
        .select('*');
      return {
        props: { data }, // will be passed to the page component as props
      }
    }
